on_actions = {
	on_startup = {
		effect = {
			every_country = {
				# Top tabs
				add_to_array = { Info_Screen_Top_Tab_Array = 1 }
				add_to_array = { Info_Screen_Top_Tab_Array = 2 }
				add_to_array = { Info_Screen_Top_Tab_Array = 3 }
				add_to_array = { Info_Screen_Top_Tab_Array = 4 }
				set_variable = { Info_Screen_Tab = 1 }
				
				# Mod development setup
				add_to_array = { Info_Screen_Mechanics_store = 1 }
				add_to_array = { Info_Screen_Mechanics_store = 2 }
				add_to_array = { Info_Screen_Mechanics_store = 3 }


				if = {
					limit = {
						FOP_has_content = yes
					}
					set_country_flag = Info_Screen_HasCustomCloseButton
				}
				if = {
					limit = {
						check_variable = { Info_Screen_array_main^num > 0 }
					}
					for_each_loop = {
						array = Info_Screen_array_main
						add_to_array = { Info_Screen_storage_array = v }
					}
				}
				
				if = {
					limit = {
						is_ai = no
					}
					set_country_flag = FOP_Info_Screen_Visible
				}
			}
		}
	}
}
