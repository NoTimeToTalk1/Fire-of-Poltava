	south_elections_countdown = {			
		activation = {
			always = no
		}
		available = {
			NOT = {
				has_country_flag = south_elections
			}
		}
		target_trigger = {
            FROM = {
				OR = {
					has_country_flag = united_dnister
					has_country_flag = united_tavria
				}
				has_country_flag = south_elections
            }
        }
		is_good = no
		days_mission_timeout = 150
		selectable_mission = no
		timeout_effect = {
			clr_country_flag = south_elections
			FROM = {
				clr_country_flag = south_elections
			}
			if = {
				limit = {
					check_variable = {DST.dnister_rates > TVR.tavria_rates}
				}
				TVR = {
					every_core_state = {
						add_core_of = DST
					}
				}
				DST = {
					annex_country = {
						target = TVR
						transfer_troops = yes
					}
				}
			}
			else_if = {
				limit = {
					check_variable = {TVR.tavria_rates > DST.dnister_rates}
				}
				DST = {
					every_core_state = {
						add_core_of = TVR
					}
				}
				TVR = {
					annex_country = {
						target = DST
						transfer_troops = yes
					}
				}
			}
			else = {
					custom_effect_tooltip = elections_not_decided
			}
		}
	}
