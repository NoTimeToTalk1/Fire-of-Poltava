		activation = {
			always = no

		}
		target_trigger = {

				}

            }

		is_good = no

		selectable_mission = no
		timeout_effect = {
			clr_country_flag = south_elections
			FROM = {
				clr_country_flag = south_elections
			}
			if = {
				limit = {
					check_variable = {BUH.buh_rates > TVR.tavria_rates}
				}
				TVR = {
					every_core_state = {
						add_core_of = BUH
					}
				}
				BUH = {
					annex_country = {
						target = TVR
						transfer_troops = yes
					}
				}
			}
			else_if = {
				limit = {
					check_variable = {TVR.tavria_rates > BUH.buh_rates}
				}
				BUH = {
					every_core_state = {
