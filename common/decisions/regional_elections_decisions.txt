		activation = {
			always = no
		}

		}
		target_trigger = {
            FROM = {

				}

            }

		is_good = no
		days_mission_timeout = 150
		selectable_mission = no
		timeout_effect = {
			clr_country_flag = south_elections
			FROM = {
				clr_country_flag = south_elections
			}
			if = {
				limit = {
					check_variable = {ROOT.south_west_rates > FROM.south_center_rates}
				}
				FROM = {
					every_core_state = {
						add_core_of = ROOT
					}
				}
				ROOT = {
					annex_country = {
						target = FROM
						transfer_troops = yes
					}
					country_event = {
						id = reg_unify.2
					}
				}
			}
			else_if = {
				limit = {
					check_variable = {ROOT.south_center_rates > FROM.south_west_rates}
				}
				FROM = {
					every_core_state = {
