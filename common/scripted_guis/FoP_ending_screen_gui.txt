scripted_gui = {
	
	FOP_Ending_Screen = {
		context_type = player_context
		window_name = "FOP_Ending_Screen"
		visible = { has_country_flag = FOP_Ending_Screen_Visible }
		dynamic_lists = {
			Top_Tabs = {
				array = Ending_Screen_Top_Tab_Array
				entry_container = "Ending_Screen_Top_Tabs_entry"
				value = tab
			}
		}
		effects = {
			tab_button_click = {
				if = {
					limit = {
						check_variable = { tab = 2 }
					}
					clear_array = Ending_Screen_array_main
					for_each_loop = {
						array = Ending_Screen_Credits_store
						add_to_array = { Ending_Screen_array_main = v }
					}
				}
				if = {
					limit = {
						NOT = {
							check_variable = { tab = Ending_Screen_Tab }
						}
					}
					set_variable = { Ending_Screen_Selected = 1 }
				}
				set_variable = { Ending_Screen_Tab = tab }
			}
			closing_button_click = {
				clr_country_flag = FOP_Ending_Screen_Visible
			}
		}
		triggers = {
			tab_button_visible = {
				if = {
					limit = {
						check_variable = { Ending_Screen_Tab = tab }
					}
					set_temp_variable = { tab_frame = 1 }
					set_temp_variable = { tab_icon_y = 9 }
				}
				else = {
					set_temp_variable = { tab_frame = 2 }
					set_temp_variable = { tab_icon_y = 14 }
				}
			}
			content_glow_visible = {
				NOT = { has_variable = Ending_Screen_Selected }
			}
			entry_title_visible = {
				NOT = {
					check_variable = { Ending_Screen_Tab = 2 }
				}
			}
			entry_desc_visible = {
				NOT = {
					check_variable = { Ending_Screen_Tab = 2 }
				}
			}
		}
		properties = {
			flag = { image = "[ROOT.GetFlag]" }
			poster = { image = "[Get_Poster]" }

			tab_button = {
				frame = tab_frame
			}
			tab_text= {
				y = tab_icon_y
			}
		}
	}
	
	FOP_Ending_Screen_Credits = {
		parent_window_name = FOP_Ending_Screen
		window_name = "FOP_Ending_Screen_Credits"
		context_type = player_context
		visible = {
			check_variable = { Ending_Screen_Tab = 2 }
		}
		dynamic_lists = {
			entry_gridbox = {
				array = Ending_Screen_array_main
				entry_container = "[Ending_Screen_Get_Entry_Container]"
				value = entry
			}
		}
		effects = {
			Content_button_click = {
				set_variable = { Ending_Screen_Selected = entry }
			}
		}
		triggers = {
			Content_button_visible = {
				if = {
					limit = {
						check_variable = { Ending_Screen_Selected = entry }
					}
					set_temp_variable = { content_frame = 2 }
				}
				else = {
					set_temp_variable = { content_frame = 1 }
				}
			}
		}
		properties = {
			Content_button = 	{ frame = content_frame }
		}
	}
}
