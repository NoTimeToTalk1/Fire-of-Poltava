first_elections_round = {
        custom_effect_tooltip = DPRelection_first_round_tt
        if = {
		    limit = {
			    check_variable = {
                    FDP_pp > DPN_pp
                }
				
				check_variable = {
                    FDP_pp > SWPD_pp
                }
			}
		    add_to_variable = { FDP_pp = DPN_pp }
			add_to_variable = { FDP_pp = SWPD_pp }
			add_to_variable = { NDPD_pp = CP_pp }
			clear_variable = CP_pp
			clear_variable = NDPD_pp_ch
			clear_variable = FDP_pp_ch
			clear_variable = DPN_pp_ch
			clear_variable = SWPD_pp_ch
			clear_variable = CP_pp_ch
			clear_variable = DPN_pp
			clear_variable = SWPD_pp
			set_variable = { opposition_leader = 2 }
			country_event = {
                id = DPRelection.20
            }
		}
		
		else_if = {
		    limit = {
			    check_variable = {
                    DPN_pp > FDP_pp
                }
				
				check_variable = {
                    DPN_pp > SWPD_pp
                }
			}
		    add_to_variable = { DPN_pp = FDP_pp }
			add_to_variable = { DPN_pp = SWPD_pp }
			add_to_variable = { NDPD_pp = CP_pp }
			clear_variable = CP_pp
			clear_variable = NDPD_pp_ch
			clear_variable = FDP_pp_ch
			clear_variable = DPN_pp_ch
			clear_variable = SWPD_pp_ch
			clear_variable = CP_pp_ch
			clear_variable = FDP_pp
			clear_variable = SWPD_pp
			set_variable = { opposition_leader = 3 }
			country_event = {
                id = DPRelection.20
            }
		} 

        else = {
		    add_to_variable = { SWPD_pp = FDP_pp }
			add_to_variable = { SWPD_pp = DPN_pp }
			add_to_variable = { NDPD_pp = CP_pp }
			clear_variable = CP_pp
			clear_variable = NDPD_pp_ch
			clear_variable = FDP_pp_ch
			clear_variable = DPN_pp_ch
			clear_variable = SWPD_pp_ch
			clear_variable = CP_pp_ch
			clear_variable = FDP_pp
			clear_variable = DPN_pp
			set_variable = { opposition_leader = 4 }
			country_event = {
                id = DPRelection.20
            }
		}
}
